    return detected

def extract_commands(text, keywords):

    text = text.lower()

    for command_name, command_info in keywords.items():
        for keyword in command_info["keywords"]:
            if keyword in text:
                # –ü–æ–∏—Å–∫ –∞—Ä–≥—É–º–µ–Ω—Ç–∞ –ø–æ—Å–ª–µ –∫–ª—é—á–µ–≤–æ–≥–æ —Å–ª–æ–≤–∞

    return found
